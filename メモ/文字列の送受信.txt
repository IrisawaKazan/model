<文字列の送受信>

(送信)

char aData[MAX_DATA] = {}; // MAX_DATAは送信側と受信側で合わせる
scanf("%s", &aData[0]);

// 送信
send(sock, &aData[0], strlen(&aData[0]) + 1 /* ヌル文字(\0)の分 */, 0);


(受信)

char aRecvData[MAX_DATA] = {};

// 受信
int nRecvByte = recv(sock, &aRecvData[0], sizeof(aRecvData), 0);

printf("%s", &aRecvData[0]);


(接続の切断)

int nRecvByte = recv(sock, ・・・);
